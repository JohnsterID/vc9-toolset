# CMake function unit tests for vc9-toolset.cmake
# Run:  cmake -B build -S test/cmake-functions && cmake --build build
# No downloads, no cross-compiler, no ISO needed — pure CMake logic tests.

cmake_minimum_required(VERSION 3.19)
project(vc9_cmake_tests NONE)

set(VC9_TOOLSET_DIR "${CMAKE_CURRENT_LIST_DIR}/../..")
include("${VC9_TOOLSET_DIR}/vc9-toolset.cmake")

# ── Helper: create mock extraction layout ──────────────────────────
function(_create_mock dest)
    file(MAKE_DIRECTORY "${dest}/VC/bin" "${dest}/VC/include" "${dest}/VC/lib"
         "${dest}/VC/lib/amd64" "${dest}/WinSDK/Include" "${dest}/WinSDK/Lib"
         "${dest}/WinSDK/Lib/x64")
    file(TOUCH "${dest}/VC/bin/cl.exe")
endfunction()

set(_mock "${CMAKE_BINARY_DIR}/_mock_vc9")
_create_mock("${_mock}")

set(_pass 0)
set(_fail 0)

# ── Test 1: default arch => x86 ───────────────────────────────────
unset(VC9_ARCH)
unset(CMAKE_C_COMPILER_TARGET)
vc9_setup("${_mock}")
if(VC9_TARGET_TRIPLE STREQUAL "i686-pc-windows-msvc")
    math(EXPR _pass "${_pass} + 1")
    message(STATUS "[PASS] default arch => i686-pc-windows-msvc")
else()
    math(EXPR _fail "${_fail} + 1")
    message(STATUS "[FAIL] default arch: expected i686, got ${VC9_TARGET_TRIPLE}")
endif()

# ── Test 2: VC9_ARCH=x64 => x86_64 triple + amd64 lib paths ──────
set(VC9_ARCH "x64")
vc9_setup("${_mock}")
if(VC9_TARGET_TRIPLE STREQUAL "x86_64-pc-windows-msvc")
    math(EXPR _pass "${_pass} + 1")
    message(STATUS "[PASS] VC9_ARCH=x64 => x86_64-pc-windows-msvc")
else()
    math(EXPR _fail "${_fail} + 1")
    message(STATUS "[FAIL] x64: expected x86_64, got ${VC9_TARGET_TRIPLE}")
endif()
list(FIND VC9_LIB_DIRS "${_mock}/VC/lib/amd64" _idx)
if(NOT _idx EQUAL -1)
    math(EXPR _pass "${_pass} + 1")
    message(STATUS "[PASS] x64 lib dirs include VC/lib/amd64")
else()
    math(EXPR _fail "${_fail} + 1")
    message(STATUS "[FAIL] x64 lib dirs missing VC/lib/amd64: ${VC9_LIB_DIRS}")
endif()
list(FIND VC9_LIB_DIRS "${_mock}/WinSDK/Lib/x64" _idx)
if(NOT _idx EQUAL -1)
    math(EXPR _pass "${_pass} + 1")
    message(STATUS "[PASS] x64 lib dirs include WinSDK/Lib/x64")
else()
    math(EXPR _fail "${_fail} + 1")
    message(STATUS "[FAIL] x64 lib dirs missing WinSDK/Lib/x64: ${VC9_LIB_DIRS}")
endif()
unset(VC9_ARCH)

# ── Test 3: all documented variables exported ──────────────────────
vc9_setup("${_mock}")
foreach(_var VC9_ROOT VC9_BIN_DIR VC9_INCLUDE_DIRS VC9_LIB_DIRS
             VC9_TARGET_TRIPLE WINSDK_INCLUDE_DIR WINSDK_LIB_DIR
             VC9_COMPAT_HEADER VC9_RUNTIME_STUBS_SRC
             VC9_CLANG_COMPILE_FLAGS VC9_CLANG_LINK_LIBS)
    if(DEFINED ${_var} AND NOT "${${_var}}" STREQUAL "")
        math(EXPR _pass "${_pass} + 1")
    else()
        math(EXPR _fail "${_fail} + 1")
        message(STATUS "[FAIL] ${_var} not exported or empty")
    endif()
endforeach()
message(STATUS "[PASS] all 11 documented variables exported (checked individually)")

# ── Test 4: compat files exist ─────────────────────────────────────
if(EXISTS "${VC9_COMPAT_HEADER}")
    math(EXPR _pass "${_pass} + 1")
    message(STATUS "[PASS] VC9_COMPAT_HEADER exists: ${VC9_COMPAT_HEADER}")
else()
    math(EXPR _fail "${_fail} + 1")
    message(STATUS "[FAIL] VC9_COMPAT_HEADER not found: ${VC9_COMPAT_HEADER}")
endif()
if(EXISTS "${VC9_RUNTIME_STUBS_SRC}")
    math(EXPR _pass "${_pass} + 1")
    message(STATUS "[PASS] VC9_RUNTIME_STUBS_SRC exists: ${VC9_RUNTIME_STUBS_SRC}")
else()
    math(EXPR _fail "${_fail} + 1")
    message(STATUS "[FAIL] VC9_RUNTIME_STUBS_SRC not found: ${VC9_RUNTIME_STUBS_SRC}")
endif()

# ── Test 9: all CRT configs include kernel32.lib ──────────────────
foreach(_crt MD MDd MT MTd)
    set(VC9_CRT "${_crt}")
    vc9_setup("${_mock}")
    list(FIND VC9_CLANG_LINK_LIBS "kernel32.lib" _idx)
    if(NOT _idx EQUAL -1)
        math(EXPR _pass "${_pass} + 1")
    else()
        math(EXPR _fail "${_fail} + 1")
        message(STATUS "[FAIL] CRT=${_crt}: missing kernel32.lib")
    endif()
endforeach()
message(STATUS "[PASS] all 4 CRT configs include kernel32.lib")

# ── Test 10: compile flags contain required entries ────────────────
set(VC9_CRT "MD")
vc9_setup("${_mock}")
string(FIND "${VC9_CLANG_COMPILE_FLAGS}" "-fms-compatibility" _pos)
if(NOT _pos EQUAL -1)
    math(EXPR _pass "${_pass} + 1")
    message(STATUS "[PASS] compile flags include -fms-compatibility")
else()
    math(EXPR _fail "${_fail} + 1")
    message(STATUS "[FAIL] compile flags missing -fms-compatibility")
endif()
string(FIND "${VC9_CLANG_COMPILE_FLAGS}" "_MSC_VER=1500" _pos)
if(NOT _pos EQUAL -1)
    math(EXPR _pass "${_pass} + 1")
    message(STATUS "[PASS] compile flags include _MSC_VER=1500")
else()
    math(EXPR _fail "${_fail} + 1")
    message(STATUS "[FAIL] compile flags missing _MSC_VER=1500")
endif()

# ── Test 11: skip re-extraction when cl.exe exists ────────────────
# vc9_setup() should not invoke extract-vc9.sh if cl.exe is present
set(_mock2 "${CMAKE_BINARY_DIR}/_mock_vc9_2")
_create_mock("${_mock2}")
set(VC9_CRT "MD")
unset(VC9_ARCH)
vc9_setup("${_mock2}")
if(VC9_ROOT STREQUAL "${_mock2}")
    math(EXPR _pass "${_pass} + 1")
    message(STATUS "[PASS] vc9_setup() skipped extraction for existing cl.exe")
else()
    math(EXPR _fail "${_fail} + 1")
    message(STATUS "[FAIL] VC9_ROOT mismatch: ${VC9_ROOT} != ${_mock2}")
endif()

# ── Summary ────────────────────────────────────────────────────────
message(STATUS "")
message(STATUS "========================================")
message(STATUS "  Results: ${_pass} passed, ${_fail} failed")
message(STATUS "========================================")
if(_fail GREATER 0)
    message(FATAL_ERROR "Some tests failed")
endif()
